16   public class DBHelper extends SQLiteOpenHelper {

18       // The database name 
19       public static final String DATABASE_NAME = "MyContacts.db"; 
20    
21       // The table "contacts" 
22       public static final String CONTACTS_TABLE_NAME    = "contacts"; 
23       public static final String CONTACTS_COLUMN_ID     = "id"; 
24       public static final String CONTACTS_COLUMN_NAME   = "name"; 
25       public static final String CONTACTS_COLUMN_EMAIL  = "email"; 
26       public static final String CONTACTS_COLUMN_STREET = "street"; 
27       public static final String CONTACTS_COLUMN_CITY   = "place"; 
28       public static final String CONTACTS_COLUMN_PHONE  = "phone"; 
29    
30       public DBHelper(Context context) { 
31           // Syntax: SQLiteOpenHelper(Context context, String name, SQLiteDatabase.CursorFactory factory, int version) 
32           // The third argument is used to allow returning subclasses of Cursor when calling query 
33           super(context, DATABASE_NAME , null, 1); 
34       } 
35    
36       /** 
37        * 
38        * @param db SQLiteDatabase to create 
39        */ 
40       @Override 
41       public void onCreate(SQLiteDatabase db) { 
42    
43           // Creating the tables - Here I will create just one table called "contacts" 
44           db.execSQL( 
45                   "create table " +  CONTACTS_TABLE_NAME + "(" + 
46                           CONTACTS_COLUMN_ID + " integer primary key, " + 
47                           CONTACTS_COLUMN_NAME   + " text, " + 
48                           CONTACTS_COLUMN_PHONE  + " text, " + 
49                           CONTACTS_COLUMN_EMAIL  + " text, " + 
50                           CONTACTS_COLUMN_STREET + " text, " + 
51                           CONTACTS_COLUMN_CITY   + " text)" 
52           ); 
53       } 
54    
55       /** 
56        * In case of upgrate, I will delete the current tables and create a new one 
57        * @param db 
58        * @param oldVersion 
59        * @param newVersion 
60        */ 
61       @Override 
62       public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) { 
63    
64           db.execSQL("DROP TABLE IF EXISTS contacts"); 
65           onCreate(db); 
66       }

68       /** 
69        * Insert a row (a new contact) into the contacts table 
70        * @param name 
71        * @param phone 
72        * @param email 
73        * @param street 
74        * @param place 
75        * @return 
76        */ 
77       public boolean insertContact  (String name, String phone, String email, String street, String place) 
78       { 
79           SQLiteDatabase db = this.getWritableDatabase(); 
80    
81           // Prepare the row to insert 
82           ContentValues contentValues = new ContentValues(); 
83    
84           contentValues.put("name", name); 
85           contentValues.put("phone", phone); 
86           contentValues.put("email", email); 
87           contentValues.put("street", street); 
88           contentValues.put("place", place); 
89    
90           // Insert the row 
91           db.insert("contacts", null, contentValues); 
92           return true; 
93       }

95       /** 
96        * Retrieving a row given a contact ID 
97        * Here, we will use the method rawQuery 
98        * @param id 
99        * @return 
100       */ 
101      public Cursor getData(int id){ 
102          SQLiteDatabase db = this.getReadableDatabase(); 
103          Cursor res =  db.rawQuery( "select * from contacts where id="+id+"", null ); 
104          return res; 
105      }

108       * Count the number of rows in a table 
109       * @return 
110       */ 
111      public int numberOfRows(){ 
112          SQLiteDatabase db = this.getReadableDatabase(); 
113   
114          return  (int) DatabaseUtils.queryNumEntries(db, CONTACTS_TABLE_NAME);

116      }

118      /** 
119       * Update the row of ID = id with these new values 
120       * @param id 
121       * @param name 
122       * @param phone 
123       * @param email 
124       * @param street 
125       * @param place 
126       * @return 
127       */ 
128      public boolean updateContact (Integer id, String name, String phone, String email, String street,String place) { 
129   
130          SQLiteDatabase db = this.getWritableDatabase(); 
131   
132          // Prepare the new values 
133          ContentValues contentValues = new ContentValues(); 
134          contentValues.put("name", name); 
135          contentValues.put("phone", phone); 
136          contentValues.put("email", email); 
137          contentValues.put("street", street); 
138          contentValues.put("place", place); 
139   
140          // run the update query 
141          db.update("contacts", contentValues, "id = ? ", new String[] { Integer.toString(id) } ); 
142          return true; 
143      } 
144   
145      /** 
146       * Delete the row of ID = id 
147       * @param id 
148       * @return 
149       */ 
150      public Integer deleteContact (Integer id) { 
151   
152          SQLiteDatabase db = this.getWritableDatabase(); 
153          return db.delete("contacts", 
154                  "id = ? ", 
155                  new String[] { Integer.toString(id) }); 
156      }

158      /** 
159       * Get the list of all contacts 
160       * @return 
161       */ 
162      public ArrayList<String> getAllContacts() 
163      { 
164          ArrayList<String> array_list = new ArrayList<String>(); 
165   
166          SQLiteDatabase db = this.getReadableDatabase(); 
167          Cursor res =  db.rawQuery( "select * from contacts", null ); 
168          res.moveToFirst(); 
169   
170          while(res.isAfterLast() == false){ 
171              array_list.add(res.getString(res.getColumnIndex(CONTACTS_COLUMN_NAME))); 
172              res.moveToNext(); 
173          } 
174          return array_list; 
175      }
